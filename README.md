			((‘’API’’)

			(‘’usability’’) 
			OR (‘‘ISO 9241’’)

		OR (‘’API us*’’) 
		AND (LIMIT-TO(SUBJAREA, ‘’COMP’’)) 
		AND (LIMIT-TO(LANGUAGE, ‘’English’’) 
			OR LIMIT-TO (LANGUAGE, ‘’French’’)
